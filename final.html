# code -1 as missing
vaccine_dat <- vaccine %>% 
  mutate(overall = ifelse(overall==-1, NA,overall),
         mmr = ifelse(mmr==-1,NA,mmr)
         ) %>%   # Oklahoma is null: double check with Aya
  mutate(year_recode = case_when(year=="null" ~"2017-18",
                                 year=="2017" ~"2017-18",
